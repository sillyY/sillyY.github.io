# 神奇的JS骚操作

  1. 使用左移运算符 << 迅速得出2的次方
1 << 2 // 4
2 << 10 // 1024

// 使用场景
a = 2e9;   // 2000000000
a << 1;    // -294967296
  1. 私用 ^ 切换变量0或1

^=是异或运算，相同取0，不同取1.

3^4 = 7
十进制转二进制: parseInt(3).toString(2) => "011", parseInt(4).toString(2) => "100"
0 1 1
1 0 0
1 1 1
二进制转十进制: parseInt(111, 2) => 7

// --- before ---
// if
if(toggle) {
    toggle = 0
}else {
    toggle = 1
}

// 三目运算符
toggle = toggle ? 0 : 1

// --- after ---
toggle ^= 1
  1. 使用 & 判断奇偶性

偶数&1 = 0 奇数&1 = 1

console.log(7&1);  // 1
console.log(8&1); // 0
  1. 使用 !! 将数字转为布尔值

下面6种值转化为布尔值时为false，其他转化都为true

  • undefined（未定义，找不到值时出现）
  • null（代表空值）
  • false（布尔值的false，字符串"false"布尔值为true）
  • 0（数字0，字符串"0"布尔值为true）
  • NaN（无法计算结果时出现，表示"非数值"；但是typeof NaN==="number"）
  • ""（双引号）或''（单引号） （空字符串，中间有空格时也是true）
console.log(!!7) // true
console.log(!!0) // false
console.log(!!-1) // true
console.log(!!0.71) // true
  1. 使用 ~~、>> 0、<< 0、>>> 0、| 0来取整 相当于 使用了Math.floor()

注意>>>不可对负数取整

console.log(~~11.71) // 11
console.log(11.71 >> 0) // 11
console.log(11.71 << 0) // 11
console.log(11.71 | 0) // 11
console.log(11.71 >>> 0) // 11
  1. 使用 ^ 来完成值交换

目前推崇ES6解构赋值

// --- before ---
let temp = a; a = b; b = temp; // 传统方案
b = [a, a = b][0] // 借助数组

// --- after ---
let a = 7
let b = 1
a ^= b
b ^= a
a ^= b

console.log(a) // 1
console.log(b) // 7

[a, b] = [b, a] // ES6， 解构赋值
  1. 使用 ^ 判断符号是否相同（正负数） (a ^ b) >= 0; // true 相同； false 不相同
(-1 ^ 11) >= 0 // false
(2 ^ 9) >= 0 // true
  1. 使用 ^ 来检查数字是否不相等
// --- before ---
if(a !== 5) {...}

// --- after ---
if(a ^ 5) {...}

9.使用 n&(n-1) === 0 判断是否为2的整数幂

var n = 24
n & (n - 1) // 16 

var n = 64
n & (n - 1) // 0
  1. 使用 A + 0.5 | 0 来替代Math.round()
var A = 24.7
A + 0.5 | 0 // => 25

var A = -24.7
A - 0.5 | 0 // => -25
  1. 使用 toString(16) 取随机字符串

.substring() 的第二个参数控制取多少位 (最多可取13位)

Math.random().toString(16).substring(2, 15);
// => "a580e0b1817b"

Math.random().toString(16).substring(2, 30);
// => "9cbbd145aa5fc"
  1. 使用 split(0)

使用数字来做为 split 的分隔条件可以节省2字节

// --- before ---
"alpha,bravo,charlie".split(",");

// --- after ---
"alpha0bravo0charlie".split(0);
  1. 使用 .link() 创建链接

可以快速创建 a 标签

// --- before ---
let b = `<a herf="www.google.com">google</a>`;

// --- after ---
let b = 'google'.link('www.google.com');
  1. 使用 Array 来重复字符
// --- before ---
for(let a = "", i = 7; i--;) a+=0

// --- after ---
let b = Array(7).join(0); // "0000000"
let c = Array(7).join('La') // "LaLaLaLaLaLa"

// ES6
let d = "0".repeat(7); // "0000000"
  1. 使用当前时间创建一个随机数

不要在快速循环中使用，因为毫秒可能没有变化；

// --- before ---
let b = 0 | Math.random() * 100

// --- after ---
let a;
a = new Date % 100; // 两位随机数
a = new Date % 1000; // 三位随机数
a = new Date % 10000; // 四位随机数...依次类推
  1. 一些可以代替 undefined 的操作
  • ""., 1.. 和 0[0]
  • void 0 // => 推崇
  1. 使用 1/0 来代替 Infinity
// --- before ---
[Infinity, -Infinity]

// --- after ---
[1/0, -1/0]
  1. 使用 Array.length = 0 来清空数组
let arr = [1, 2, 3]
arr.length = 0

arr // => []
  1. 使用 Array.slice(0) 实现数组浅拷贝
let arr = [0, 1, 2, 3]

arr.slice(0) // => [0, 1, 2, 3]
  1. 使用 !+"\v1" 快速判断 IE8 以下的浏览器
var isIE8 = !+"\v1" 

// Chrome false
// IE9(10, 11) false
// IE8(7, 6, 5) true
  1. for循环条件的简写
// --- before ---
for(let i = 0; i < arr.length; i++) {...}

// --- after ---
for(let i = arr.length; i--;) {...} // 注意 i--后面的分号不可漏
